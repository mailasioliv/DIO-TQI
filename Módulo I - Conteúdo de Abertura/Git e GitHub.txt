cd = navegar nas pastas
cd.. = voltar para nível anterior
echo > 'nome e extensão do arquivo' = criar novo arquivo
ls = listar arquivos
ls -a = listar arquivos mostrando arquivos ocultos
mkdir = criar pasta 
mv 'nome do arquivo' ./endereço da pasta = alterar local do arquivo
git init = inicializar repositório
git add * = aterar todos arquivos para staged
git add 'nome do arquivo ou nome da pasta/' = atualizar status para staged
git status = verificar se existe alguma pendência nos arquivos
git commit -m "mensagem de alteração" = identificar alterações realizadas no repositório
git config --list = listar configurações do git
git config --global 'campo para preencher' = configurar e-mail e nome de usuário do git
git config --global -- unset 'campo para alterar' = alterar dados de configurações do git
git remote add origin 'link https' = adicionar a origem dos arquivos locais para o remoto (GitHub)
git remote -v = listar os repositórios remotos cadastrados
git push orign 'master ou main' = enviar para o GitHub
